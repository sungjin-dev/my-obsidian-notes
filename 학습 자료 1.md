
# 🛸 ESP32-CAM ➔ Flask ➔ YOLO 실시간 데이터 수집, 학습 및 객체 검출 파이프라인

본 가이드는 ESP32-CAM을 클라이언트로 사용하여 촬영된 프레임을 Flask 서버로 전송하고, 이를 OpenCV로 가공하여 YOLO 학습 데이터로 수집한 후, 최종 학습 모델을 통해 실시간으로 경계 상자(Bounding Box)를 그려 웹 스트리밍하는 전체 과정을 담고 있습니다.

## 📌 전체 시스템 아키텍처 흐름도

```
[ 1. ESP32-CAM ]
      │ (Wi-Fi를 통한 HTTP POST 전송, JPEG 포맷)
      ▼
[ 2. Flask 서버 (/upload 엔드포인트) ] ───► 수집 모드 활성화 시: /dataset/raw_images/ 에 저장
      │
      ├─► OpenCV 이미지 디코딩 & 프레임 메모리 적재
      │
      ▼ (학습 완료 후)
[ 3. YOLOv8 실시간 추론 (Inference) ] ───► 프레임 위에 Bounding Box 및 Class Label 렌더링
      │
      ▼ (Multipart MJPEG 스트림 전송)
[ 4. 웹 브라우저 (/video_feed 엔드포인트) ] ───► 사용자 화면에 실시간 결과 출력
```

## 1단계: ESP32-CAM 소스 코드 (클라이언트)

 **AI-Thinker ESP32-CAM** 모듈을 기준. 카메라로 찍은 JPEG 이미지 바이너리를 설정된 Flask 서버의 `/upload` 주소로 `HTTP POST` 방식을 통해 전송

```
#include "esp_camera.h"
#include <WiFi.h>
#include <HTTPClient.h>

// **1. WiFi 및 서버 설정**
const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";
const char* serverName = "http://192.168.x.x:5000/upload"; // 

// **2. AI-Thinker 카메라 핀 매핑 설정**
#define PWDN_GPIO_NUM     32
#define RESET_GPIO_NUM    -1
#define XCLK_GPIO_NUM      0
#define SIOD_GPIO_NUM     26
#define SIOC_GPIO_NUM     27
#define Y9_GPIO_NUM       35
#define Y8_GPIO_NUM       34
#define Y7_GPIO_NUM       39
#define Y6_GPIO_NUM       36
#define Y5_GPIO_NUM       21
#define Y4_GPIO_NUM       19
#define Y3_GPIO_NUM       18
#define Y2_GPIO_NUM        5
#define VSYNC_GPIO_NUM    25
#define HREF_GPIO_NUM     23
#define PCLK_GPIO_NUM     22

void setup() {
  Serial.begin(115200);
  
  // WiFi 연결
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi Connected!");

  // 카메라 설정 구조체 정의
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;

  // YOLO 학습용 및 실시간 감지용으로 적합한 해상도 설정 (SVGA 또는 VGA)
  if(psramFound()){
    config.frame_size = FRAMESIZE_VGA; // 640x480
    config.jpeg_quality = 12; // 0-63 (낮을수록 고화질)
    config.fb_count = 2;
  } else {
    config.frame_size = FRAMESIZE_CIF; // 400x296
    config.jpeg_quality = 12;
    config.fb_count = 1;
  }

  // 카메라 초기화
  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("Camera init failed with error 0x%x", err);
    return;
  }
}

void loop() {
  camera_fb_t * fb = NULL;
  fb = esp_camera_fb_get();
  if(!fb) {
    Serial.println("Camera capture failed");
    delay(1000);
    return;
  }

  // HTTP POST 요청 전송
  if(WiFi.status() == WL_CONNECTED){
    HTTPClient http;
    http.begin(serverName);
    http.addHeader("Content-Type", "image/jpeg");
    
    // 바이트 배열을 POST 바디로 전송
    int httpResponseCode = http.POST(fb->buf, fb->len);
    
    if (httpResponseCode > 0) {
      Serial.printf("Response code: %d\n", httpResponseCode);
    } else {
      Serial.printf("Error on sending POST: %s\n", http.errorToString(httpResponseCode).c_str());
    }
    http.end();
  }
  
  // 프레임 버퍼 반환 및 전송 주기(딜레이) 조절
  esp_camera_fb_return(fb);
  delay(100); // 약 10 FPS 전송 목표
}
```

## 2단계: Flask 서버 기본 구축 (OpenCV 데이터 수집 및 웹 스트리밍)

이 서버는 ESP32-CAM이 쏜 이미지를 실시간으로 받아서, 메모리에 보관하면서 웹 브라우저에 뿌려줌. **YOLO 모델 학습용 이미지를 수집**할 수 있도록 파일로 저장하는 기능도 포함.

```
# app.py
from flask import Flask, request, Response, render_template_string
import cv2
import numpy as np
import os
import time

app = Flask(__name__)

# 글로벌 변수로 현재 수신한 프레임을 보관
current_frame = None

# 1. 학습용 이미지 저장을 위한 디렉토리 구조 생성
SAVE_DIR = "dataset/raw_images"
if not os.path.exists(SAVE_DIR):
    os.makedirs(SAVE_DIR)

# 수집 모드 플래그 (True 설정 시 들어오는 프레임을 파일로 저장)
COLLECT_DATA_MODE = False

@app.route('/upload', methods=['POST'])
def upload():
    global current_frame
    try:
        # ESP32-CAM이 보낸 바이너리 데이터 수신
        data = request.data
        if not data:
            return "No data", 400
            
        # 바이트 어레이를 numpy 배열로 변환 후 OpenCV로 디코딩 (JPEG -> BGR 이미지)
        nparr = np.frombuffer(data, np.uint8)
        img = cv2.imdecode(nparr, cv2.IMREAD_COLOR)
        
        if img is not None:
            current_frame = img
            
            # [학습 데이터 수집 모드] 활성화 시 프레임을 고유 파일명으로 저장
            if COLLECT_DATA_MODE:
                timestamp = int(time.time() * 1000)
                file_path = f"{SAVE_DIR}/frame_{timestamp}.jpg"
                cv2.imwrite(file_path, img)
                print(f"saved: {file_path}")
                
            return "Frame Received", 200
    except Exception as e:
        print(f"Error: {e}")
        return str(e), 500
    return "Failed to process", 400

# MJPEG 포맷으로 스트리밍 제너레이터 함수 구현
def gen_frames():
    global current_frame
    while True:
        if current_frame is not None:
            # 프레임을 JPEG 바이트로 인코딩하여 웹 브라우저에 던져줌
            ret, buffer = cv2.imencode('.jpg', current_frame)
            frame_bytes = buffer.tobytes()
            yield (b'--frame\r\n'
                   b'Content-Type: image/jpeg\r\n\r\n' + frame_bytes + b'\r\n')
        time.sleep(0.05)

@app.route('/video_feed')
def video_feed():
    # 웹 브라우저가 끊어지지 않고 계속 스트림을 수신하게 함
    return Response(gen_frames(), mimetype='multipart/x-mixed-replace; boundary=frame')

# 간단한 모니터링 페이지 HTML 랜더링
@app.route('/')
def index():
    return render_template_string("""
        <html>
            <head><title>ESP32-CAM Real-time Feed</title></head>
            <body style="background-color: #1e293b; color: white; text-align: center; font-family: sans-serif;">
                <h1>ESP32-CAM 실시간 모니터링 및 수집기</h1>
                <div style="margin: 20px;">
                    <img src="/video_feed" style="border: 5px solid #a4d65e; border-radius: 12px; width: 640px; height: 480px;"/>
                </div>
                <p>수집 모드 상태: <strong>{{ "활성화" if collect else "비활성화" }}</strong></p>
            </body>
        </html>
    """, collect=COLLECT_DATA_MODE)

if __name__ == '__main__':
    # 외부 기기(ESP32)에서 접속할 수 있도록 host='0.0.0.0' 포트 5000번 오픈
    app.run(host='0.0.0.0', port=5000, debug=False)
```

## 3단계: YOLOv8 학습용 데이터 준비 및 라벨링 (Labeling)

Flask 서버 수집 기능(`COLLECT_DATA_MODE = True`)을 켜두고 ESP32-CAM을 움직이며 다양한 환경에서 수집한 사진들을 가지고 YOLO 학습 설계

### **1. 라벨링 도구 다운로드 및 실행**

가장 간편하고 강력한 웹/로컬 도구인 **Roboflow** 또는 **LabelImg**를 이용합니다.

- **로컬 설치형(LabelImg):**
    
    ```
    pip install labelImg
    labelImg
    ```
    
- `Open Dir`을 통해 `dataset/raw_images` 폴더를 열고, 학습 데이터 포맷을 **YOLO**로 설정한 후 네모 상자(Bounding Box)를 그려 라벨(예: `leak`, `fire`, `person` 등)을 부여
    
- 완료하면 각 이미지 이름과 매칭되는 `.txt` 파일(라벨 파일)들이 생성됩니다.

### **2. 학습을 위한 디렉토리 구조 배치**

수집 및 라벨링이 끝난 데이터셋을 학습 엔진이 인식할 수 있도록 아래와 같이 폴더 트리를 정리 (학습용 `train`과 검증용 `val`로 보통 8:2 비율로 쪼개어 배분)

```

  ├── train/
  │     ├── images/          # 학습용 .jpg 이미지들
  │     └── labels/          # 학습용 .txt 라벨들
  └── val/
        ├── images/          # 검증용 .jpg 이미지들
        └── labels/          # 검증용 .txt 라벨들
```

### **3. `data.yaml`** 파일 작성


## 4단계: YOLO 모델 학습 (Training)

라벨링된 데이터셋을 가지고 신경망 학습을 수행. 고사양 외장 GPU(NVIDIA)가 탑재된 PC나 Google Colab 환경을 권장.

현시점에서 가장 빠르고 높은 정확도를 자랑하는 **Ultralytics YOLOv8/v11** 패키지를 기반으로 설계.

### **1. 관련 라이브러리 설치**

### **2. 학습 파이썬 스크립트 작성 및 실행**

```
# train_yolo.py
from ultralytics import YOLO

def main():
    # 1. 사전 학습된 나노(Nano) 모델을 불러옵니다. (가장 가벼워서 실시간 임베디드 스트림에 최적)
    model = YOLO('yolov8n.pt') 

    # 2. 커스텀 데이터셋으로 모델 학습을 수행합니다.
    results = model.train(
        data='dataset/data.yaml',   # YAML 설정 파일 경로
        epochs=50,                  # 에폭 수 (데이터셋이 적으면 50~100선 권장)
        imgsz=640,                  # 입력 이미지 크기 (VGA 급 이미지)
        batch=16,                   # 배치 크기
        device=0,                   # GPU 번호 (CPU로 돌릴 경우 'cpu' 입력)
        project='custom_drone',     # 학습 결과 저장 폴더명
        name='v8_model'             # 세부 세션명
    )
    
    print("학습이 완료되었습니다. 결과 파일은 'custom_drone/v8_model/weights/best.pt'에 저장되었습니다.")

if __name__ == '__main__':
    main()
```

## 5단계: Flask + 학습된 YOLO 실시간 객체 탐색 및 라벨링 렌더링

마지막 단계입니다! 4단계 학습을 통해 성공적으로 저장된 최상의 가중치 파일인 `best.pt`를 Flask 서버 내부로 로드합니다.

ESP32-CAM으로부터 전달받는 원본 프레임 마다 `best.pt` 모델의 추론(Inference) 연산을 수행하여 라벨명과 경계상자를 OpenCV 함수로 직접 렌더링하여 화면에 리다이렉트합니다.

```
# app_yolo_inference.py
from flask import Flask, request, Response, render_template_string
from ultralytics import YOLO
import cv2
import numpy as np
import time

app = Flask(__name__)

# 글로벌 변수로 영상 프레임 관리
processed_frame = None

# 1. 4단계에서 학습 완료된 커스텀 YOLO 가중치 모델 로드
# 만약 가중치 경로가 다르면 알맞게 매칭해 주어야 합니다.
YOLO_MODEL_PATH = "custom_drone/v8_model/weights/best.pt" 
model = YOLO(YOLO_MODEL_PATH)

@app.route('/upload', methods=['POST'])
def upload():
    global processed_frame
    try:
        data = request.data
        if not data:
            return "No data", 400
            
        # ESP32 이미지 디코딩
        nparr = np.frombuffer(data, np.uint8)
        img = cv2.imdecode(nparr, cv2.IMREAD_COLOR)
        
        if img is not None:
            # 2. 이미지 프레임을 YOLO 모델에 투입하여 추론 진행 (객체 탐색)
            # stream=True 옵션을 주면 대용량 영상 메모리 누수가 발생하지 않습니다.
            results = model(img, stream=True, conf=0.5) # 정확도(신뢰도) 임계값 50% 지정
            
            # 3. 탐색된 결과를 바탕으로 OpenCV를 이용해 이미지 위에 그림 그리기
            for r in results:
                boxes = r.boxes
                for box in boxes:
                    # Bounding Box 좌표 얻기 (Left, Top, Right, Bottom)
                    x1, y1, x2, y2 = box.xyxy[0]
                    x1, y1, x2, y2 = int(x1), int(y1), int(x2), int(y2)
                    
                    # 클래스 인덱스 및 신뢰도 점수 획득
                    cls = int(box.cls[0])
                    conf = float(box.conf[0])
                    class_name = model.names[cls]
                    
                    # 4. 실시간 경계 상자 렌더링 (안정감 있는 로열 블루 색상 적용)
                    cv2.rectangle(img, (x1, y1), (x2, y2), (161, 71, 13), 3) # BGR 순서이므로 로열 블루 색상 매칭
                    
                    # 라벨 텍스트 가시화
                    label_text = f"{class_name} {conf:.2f}"
                    cv2.putText(img, label_text, (x1, max(y1 - 10, 15)), 
                                cv2.FONT_HERSHEY_SIMPLEX, 0.6, (161, 71, 13), 2)
            
            # 최종 연산 및 변환 처리가 완료된 이미지를 글로벌 변수에 전달
            processed_frame = img
            return "Frame Annotated Successfully", 200
            
    except Exception as e:
        print(f"Error on upload & inference: {e}")
        return str(e), 500
    return "Process Failed", 400

# MJPEG 변환 제너레이터 구현
def gen_processed_frames():
    global processed_frame
    while True:
        if processed_frame is not None:
            ret, buffer = cv2.imencode('.jpg', processed_frame)
            frame_bytes = buffer.tobytes()
            yield (b'--frame\r\n'
                   b'Content-Type: image/jpeg\r\n\r\n' + frame_bytes + b'\r\n')
        time.sleep(0.04) # 대략 25fps 주기로 송출 대기

@app.route('/video_feed')
def video_feed():
    return Response(gen_processed_frames(), mimetype='multipart/x-mixed-replace; boundary=frame')

@app.route('/')
def index():
    return render_template_string("""
        <html>
            <head><title>ESP32-CAM YOLOv8 Detection</title></head>
            <body style="background-color: #0f172a; color: white; text-align: center; font-family: sans-serif; padding-top: 50px;">
                <h1 style="color: #a4d65e;">실시간 AI 해상 위험구역 감지 시스템</h1>
                <p style="color: #94a3b8;">ESP32-CAM ➔ Flask 서버 ➔ YOLOv8 실시간 연동 가동 중</p>
                <div style="margin-top: 30px;">
                    <img src="/video_feed" style="border: 6px solid #a4d65e; border-radius: 16px; width: 640px; height: 480px; box-shadow: 0 10px 30px rgba(0,0,0,0.5);"/>
                </div>
            </body>
        </html>
    """)

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=False)
```
